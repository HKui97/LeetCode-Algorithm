输入：

     4
   /   \
  2     7
 / \     / \
1   3 6   9
输出：

      4
   /     \
  7       2
 / \      / \
9   6  3   1

class Solution {
    public TreeNode invertTree(TreeNode root) {
        TreeNode temp = new TreeNode(0);
        if(root ==null) return root;
        else{
            temp = invertTree(root.right);
            root.right =invertTree(root.left);
            root.left = temp;
        }
        return root;
    }
}
